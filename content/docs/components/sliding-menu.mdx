## Anatomy
Import all parts and use them together

```tsx
import {
	SlidingMenu,
	SlidingMenuTrigger,
	SlidingMenuContent,
	SlidingMenuPortal,
	SlidingMenuPage,
	SlidingMenuItem,
	SlidingMenuSeperator,
} from "@/components/sliding-menu";
```
```tsx
<SlidingMenu>
  <SlidingMenuTrigger></SlidingMenuTrigger>
  <SlidingMenuPortal>
    <SlidingMenuContent>
      <SlidingMenuItem>
        <SlidingMenuSeperator />
      </SlidingMenuItem>
    </SlidingMenuContent>
  </SlidingMenuPortal>
</SlidingMenu>
```

### useSlidingMenu
Hook to access the sliding menu context. Must be used within a `SlidingMenu` component.

Returns:
```tsx
{
  currentPage: string;
  navigateToPage: (pageId: string) => void;
  navigateBack: () => void;
  pageHistory: string[];
  isOpen: boolean;
  setIsOpen: (open: boolean) => void;
}
```
