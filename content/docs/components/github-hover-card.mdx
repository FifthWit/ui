## Usage
```tsx
import {
	GhCardImage,
	GhCardRoot,
	GhCardTitle,
	GhCardLabel,
	GhCardHeader,
	GhCardSubHeader,
	GhCardContent,
} from "@//components/github-hover-card";
```

```tsx
					Figma gains 100x users after mandating anime girlfriends for all users

				<GhCardSubHeader>Read More</GhCardSubHeader>
			</GhCardContent>
		</GhCardRoot>
	);
}
```
